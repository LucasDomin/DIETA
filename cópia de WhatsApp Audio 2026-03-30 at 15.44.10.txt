    <script>
        // Dados das refeições (exemplo para hoje, variam por dia)
        const meals = {
            domi: [
                { icon: '☕', name: 'Café Pré-Treino', time: '06:30', cal: 400, weight: '200g', desc: '2 ovos + 1 fatia pão integral + café preto' },
                { icon: '🥤', name: 'Pós-Treino', time: '11:00', cal: 300, weight: '1 scoop', desc: 'Whey + banana + 20g castanhas' },
                { icon: '🍽️', name: 'Almoço Empresa', time: '12:30', cal: 600, weight: '400g', desc: 'Frango grelhado + arroz + salada (controlar porção)' },
                { icon: '🥪', name: 'Lanche 16h', time: '16:00', cal: 250, weight: '50g', desc: '1 fatia pão + queijo light + café' },
                { icon: '🍗', name: 'Jantar', time: '19:30', cal: 500, weight: '300g', desc: 'Carne magra + batata doce + vegetais' },
                { icon: '🧀', name: 'Pós-Treino Noturno', time: '22:00', cal: 150, weight: '200g', desc: 'Iogurte grego + paçoca fit' }
            ],
            nath: [
                { icon: '☕', name: 'Café Casa', time: '06:50', cal: 300, weight: '150g', desc: 'Ovo + aveia + morango' },
                { icon: '🥛', name: 'Lanche Escola', time: '09:50', cal: 200, weight: '150g', desc: 'Iogurte grego + 15g castanhas' },
                { icon: '🍽️', name: 'Almoço', time: '13:00', cal: 450, cal: 450, weight: '350g', desc: 'Peito frango + salada + batata doce pequena' },
                { icon: '🍌', name: 'Pré-Treino', time: '17:00', cal: 200, weight: '1 un', desc: 'Banana + whey scoop' },
                { icon: '🥗', name: 'Jantar Pós-Treino', time: '20:30', cal: 400, weight: '250g', desc: 'Ovos + vegetais refogados + quinoa' }
            ]
        };

        let data = JSON.parse(localStorage.getItem('dietData')) || { domi: { cal: 0, water: 0, history: [] }, nath: { cal: 0, water: 0, history: [] } };
        const targets = { domi: { cal: 2200, water: 3500 }, nath: { cal: 1550, water: 2500 } }; // ml para água

        function switchTab(profile) {
            document.querySelectorAll('.content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.getElementById(profile).classList.add('active');
            event.target.classList.add('active');
            loadProfile(profile);
        }

        function loadProfile(profile) {
            const mealsEl = document.getElementById(profile + '-meals');
            mealsEl.innerHTML = '';
            meals[profile].forEach((meal, i) => {
                const checked = data[profile].checked ? data[profile].checked[i] : false;
                const cal = checked ? meal.cal : 0;
                data[profile].cal = (data[profile].cal || 0) + (checked ? 0 : cal); // Atualiza soma se checked
                mealsEl.innerHTML += `
                    <div class="meal">
                        <div class="meal-icon">${meal.icon}</div>
                        <div class="meal-info">
                            <div class="meal-name">${meal.name}</div>
                            <div class="meal-time">🕐 ${meal.time}</div>
                            <div class="meal-weight">⚖️ ${meal.weight}</div>
                            <div class="meal-desc">${meal.desc}</div>
                        </div>
                        <div class="meal-actions">
                            <input type="number" id="${profile}-weight-${i}" placeholder="Peso g" value="${checked ? meal.weight : ''}">
                            <button class="check-btn ${checked ? 'checked' : ''}" onclick="toggleMeal('${profile}', ${i}, ${meal.cal})">
                                ${checked ? '✅' : '⭕'}
                            </button>
                        </div>
                    </div>
                `;
            });
            updateStats(profile);
            loadHistory(profile);
        }

        function toggleMeal(profile, index, cal) {
            data[profile].checked = data[profile].checked || [];
            data[profile].checked[index] = !data[profile].checked[index];
            data[profile].cal += data[profile].checked[index] ? cal : -cal;
            localStorage.setItem('dietData', JSON.stringify(data));
            loadProfile(profile);
        }

        function addWater(profile, ml) {
            data[profile].water += ml;
            localStorage.setItem('dietData', JSON.stringify(data));
            loadProfile(profile);
            // Salva no histórico diário
            const today = new Date().toDateString();
            if (!data[profile].history.find(h => h.date === today)) {
                data[profile].history.push({ date: today, cal: data[profile].cal, water: data[profile].water });
            }
        }

        function updateStats(profile) {
            const calEl = document.getElementById(profile + '-cal');
            const calBar = document.getElementById(profile + '-cal-bar');
            const waterEl = document.getElementById(profile + '-water');
            const waterBar = document.getElementById(profile + '-water-bar');
            const pctCal = Math.min((data[profile].cal / targets[profile].cal) * 100, 100);
            const pctWater = Math.min((data[profile].water / targets[profile].water) * 100, 100);
            calEl.textContent = `${Math.round(data[profile].cal)} / ${targets[profile].cal}`;
            waterEl.textContent = `${(data[profile].water / 1000).toFixed(1)} / ${(targets[profile].water / 1000).toFixed(1)}L`;
            calBar.style.width = pctCal + '%';
            waterBar.style.width = pctWater + '%';
        }

        function loadHistory(profile) {
            const histEl = document.getElementById(profile + '-history');
            histEl.innerHTML = data[profile].history.slice(-7).map(h => 
                `<div class="history-item">${h.date}: ${Math.round(h.cal)}kcal | ${(h.water/1000).toFixed(1)}L</div>`
            ).join('') || '<div>Nenhum histórico ainda. Comece a marcar!</div>';
        }

        // Inicializa
        loadProfile('domi');

        // Reset diário automático (simples: botão manual por enquanto)
        function resetDay(profile) {
            data[profile].cal = 0;
            data[profile].water = 0;
            data[profile].checked = [];
            localStorage.setItem('dietData', JSON.stringify(data));
            loadProfile(profile);
        }
    </script>
